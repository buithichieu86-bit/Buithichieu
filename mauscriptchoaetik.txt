-- Server_Admin (Script)
-- Danh sách admin theo UserId (chỉ định rõ ai có quyền)
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local remotes = ReplicatedStorage:WaitForChild("Remotes")
local Remote_YeuCau = remotes:WaitForChild("Remote_YeuCau")

-- Sửa danh sách Admin theo UserId của bạn
local DanhSachAdmin = {
    [12345678] = true, -- thay bằng UserId thật nếu muốn
}

Remote_YeuCau.OnServerEvent:Connect(function(nguoiChoi, hanhDong, ...)
    if not DanhSachAdmin[nguoiChoi.UserId] then
        return -- không có quyền thực thi lệnh admin
    end

    if hanhDong == "teleport" then
        local tenDich = ...
        local dich = game.Players:FindFirstChild(tenDich)
        if dich and nguoiChoi.Character and dich.Character then
            local hrpNguoi = nguoiChoi.Character:FindFirstChild("HumanoidRootPart")
            local hrpDich = dich.Character:FindFirstChild("HumanoidRootPart")
            if hrpNguoi and hrpDich then
                hrpNguoi.CFrame = hrpDich.CFrame + Vector3.new(2,0,0)
            end
        end
    elseif hanhDong == "datien" then
        local tenNguoi, soTien = ...
        local mucTieu = game.Players:FindFirstChild(tenNguoi)
        if mucTieu then
            mucTieu:SetAttribute("Tien", tonumber(soTien) or 0)
        end
    end
end)
